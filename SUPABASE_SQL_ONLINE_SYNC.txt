-- Crea una única agenda online compartida entre PC y móvil.

create table if not exists agenda_online (
  id text primary key,
  datos jsonb not null default '{}'::jsonb,
  updated_at timestamptz not null default now()
);

alter table agenda_online disable row level security;

insert into agenda_online (id, datos, updated_at)
values ('main', '{}'::jsonb, now())
on conflict (id) do nothing;

select id, updated_at, jsonb_array_length(coalesce(datos->'citas','[]'::jsonb)) as numero_citas
from agenda_online
where id='main';
