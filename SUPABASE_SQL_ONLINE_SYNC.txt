-- Esto crea la agenda online compartida entre PC y teléfono.

create table if not exists agenda_online (
  id text primary key,
  datos jsonb not null,
  updated_at timestamptz not null default now()
);

alter table agenda_online disable row level security;

insert into agenda_online (id, datos, updated_at)
values ('main', '{}'::jsonb, now())
on conflict (id) do nothing;

select id, updated_at from agenda_online;
