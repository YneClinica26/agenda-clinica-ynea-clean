-- Esto crea la agenda online única que sincroniza PC y móvil.

create table if not exists agenda_online (
  id text primary key,
  datos jsonb not null default '{}'::jsonb,
  updated_at timestamptz not null default now()
);

alter table agenda_online disable row level security;

grant usage on schema public to anon;
grant select, insert, update, delete on table agenda_online to anon;
grant select, insert, update, delete on table agenda_online to authenticated;

insert into agenda_online (id, datos, updated_at)
values ('main', '{}'::jsonb, now())
on conflict (id) do nothing;

select
  id,
  updated_at,
  jsonb_array_length(coalesce(datos->'citas','[]'::jsonb)) as numero_citas,
  jsonb_array_length(coalesce(datos->'pacientes','[]'::jsonb)) as numero_pacientes
from agenda_online
where id='main';
